add_namespace = ACW

###start america
country_event = {
	id = ACW.1
	title = ACW.1.t
	desc = ACW.1.d
	picture = GFX_report_event_soviet_soldiers_tank

	fire_only_once = yes

	trigger = {
		tag = USA
        DATE > 1937.4.29
		has_global_flag = uac_leave
		has_global_flag = ams_leave
	}
	mean_time_to_happen = { days = 1 }

	option = {
		name = ACW.1.a
		ai_chance = { factor = 100}
		declare_war_on = {
		    target = UAC
		    type = annex_everything
		}
		declare_war_on = {
		    target = AMS
		    type = annex_everything
		}
		AMS = {
		    declare_war_on = {
		        target = UAC
		        type = annex_everything
		    }
		}
		hidden_effect = {
			news_event = { id = news.2 hours = 6 }
			set_global_flag = american_civil_war
		}
	}
}

country_event = {
	id = ACW.4
	title = ACW.4.t
	desc = ACW.4.d
	picture = GFX_report_event_soviet_soldiers_tank

	fire_only_once = yes
	
	trigger = {
		tag = USA
		has_war_with = UAC
		has_war_with = AMS
		OR = {	
			check_variable = {
    			var = USA.virginia_loyalty
    			value = USA.miltia
    			compare = equals
        	}
			check_variable = {
    			var = USA.virginia_loyalty
    			value = USA.miltia
    			compare = greater_than
        	}
		}
	}
	mean_time_to_happen = { days = 1 }

	option = {
		name = ACW.4.a
		ai_chance = { factor = 100}
	    362 = {
			create_unit = {
				division = "name = \"Virginia Militia\" division_template = \"Garrison Brigade\" start_experience_factor = 0.1 start_equipment_factor = 0.5" 
				owner = USA
				allow_spawning_on_enemy_provs = yes
			}	
		}
	}
}
